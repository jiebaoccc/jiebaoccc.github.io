【id175】B. DS二叉树--赫夫曼树解码

Oct 31, 2024
2 minute read

题目描述

已知赫夫曼编码算法和程序，在此基础上进行赫夫曼解码

可以增加一个函数：int Decode(const string codestr, char txtstr[]);//输入编码串codestr，输出解码串txtstr

该方法如果解码成功则返回1，解码失败则返回-1，本程序增加宏定义ok表示1，error表示-1

赫夫曼解码算法如下：

定义指针p指向赫夫曼树结点，指针i指向编码串，定义ch逐个读取编码串的字符

初始操作包括读入编码串str，设置p指向根结点，i为0表示指向串首，执行以下循环：

1）取编码串的第i个字符放入ch

2）如果ch是字符0，则p跳转到左孩子；如果ch是字符1，则p跳转到右孩子；如果ch非0非1，表示编码串有错误，报错退出

3）如果p指的结点是叶子，输出解码字符，p跳回根结点，i++，跳步骤1

4）如果p指的结点不是叶子且i未到编码串末尾，i++，跳步骤1

5）如果p指的结点不是叶子且i到达编码串末尾，报错退出

当i到达编码串末尾，解码结束。

输入

第一行先输入n，表示有n个叶子

第二行输入n个权值，权值全是小于1万的正整数 第三行输入n个字母，表示与权值对应的字符 第四行输入k，表示要输入k个编码串 第五行起输入k个编码串

输出

每行输出解码后的字符串，如果解码失败直接输出字符串“error”，不要输出部分解码结果

IO模式

本题IO模式为标准输入/输出(Standard IO)，你需要从标准输入流中读入数据，并将答案输出至标准输出流中。

输入样例

5
15 4 4 3 2
A B C D E
3
11111
10100001001
00000101100

输出样例

AAAAA
ABEAD
error

解题思路

首先创建哈夫曼树，同id176，加入解码函数

具体代码

#include<bits/stdc++.h>
using namespace std;

class HuffNode {
public:
	int weight;
	char data;
	int parent;
	int leftchild;
	int rightchild;
	HuffNode() { weight = parent = leftchild = rightchild =0; }
};

class HuffTree {
private:
	HuffNode* h;
	int len;
	int n;

public:
	void Create_tree(int n, vector<int> wt, vector<char> data)
	{
		len = 2 * n - 1;
		this->n = n;
		h = new HuffNode[len+1];

		for (int i = 1; i <= n; i++)
		{
			h[i].weight = wt[i];
			h[i].data = data[i];
		}

		create();
	}

	void create()
	{
		for (int i = n + 1; i <= len; i++)
		{
			int n1, n2;
			FindMin(i-1,n1, n2);
			h[n1].parent = i;
			h[n2].parent = i;
			h[i].leftchild = n1;
			h[i].rightchild = n2;
			h[i].weight = h[n1].weight + h[n2].weight;
		}
	}

	void FindMin(int pos,int& n1, int& n2)
	{
		int min1 = 10000, min2 = 10000;
		for (int i = 1; i <= pos; i++)
		{
			if (min1 > h[i].weight && !h[i].parent)
			{
				min2 = min1;
				min1 = h[i].weight;
				n2 = n1;
				n1 = i;
			}
			else if (min2 > h[i].weight && !h[i].parent)
			{
				min2 = h[i].weight;
				n2 = i;
			}
		}
	}

	int Decode(string codestr, vector<char>& ans)
	{
		int tem = len;
		while (1)
		{
			if (codestr.empty())
			{
				if ((h[tem].leftchild || h[tem].rightchild) && tem != len)
					return -1;
				else
					return 1;
			}

			if (codestr[0] == '0')
			{
				tem = h[tem].leftchild;
				codestr.erase(0,1);
			}
			else if (codestr[0] == '1')
			{
				tem = h[tem].rightchild;
				codestr.erase(0,1);
			}

			if (!h[tem].leftchild && !h[tem].rightchild)
			{
				ans.push_back(h[tem].data);
				tem = len;
			}
		}
	}
};

int main()
{
	int n;
	cin >> n;
	vector<int> weight(n+1);
	vector<char> data(n+1);
	for (int i = 1; i <= n; i++)
		cin >> weight[i];
	for (int i = 1; i <= n; i++)
		cin >> data[i];

	HuffTree t,tem;
	t.Create_tree(n,weight,data);
	
	int tag;
	cin >> tag;
	while (tag--)
	{
		string s;
		vector<char> ans;
		cin >> s;
		int val = t.Decode(s,ans);
		if (val == -1)
			cout << "error" << endl;
		else
		{
			for (int i = 0; i < ans.size(); i++)
				cout << ans[i];
			cout << endl;
		}
	}
	return 0;
}
