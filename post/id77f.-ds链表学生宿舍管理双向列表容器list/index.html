【id77】F. DS链表—学生宿舍管理（双向列表容器List）

Sep 12, 2024
3 minute read

题目描述

​ 假设某校有20间宿舍，宿舍编号101，102，…,120。每间只住一名学生。初始部分宿舍已用。用两个链表（已用宿舍链表和可用宿舍链表）维护宿舍的管理，实现宿舍分配、宿舍交回。

​ 约定已用宿舍链表按宿舍号升序链接。初始可用宿舍链表也按宿舍号升序链接。

​ 宿舍分配从可用宿舍链表中摘取第一间宿舍分配给学生。学生交回的宿舍挂在可用宿舍链表最后。

​ 备注：使用list容器或静态链表。不用考虑宿舍分配和交回不成功的情况。

输入

​ 初始宿舍状态，第一行输入n，表示已用宿舍n间

​ 后跟n行数据，每行格式为：学生姓名 宿舍号

​ 操作次数m，后跟m行操作，操作格式如下：

​ assign 学生 //为学生分配宿舍，从可用宿舍链表头摘取一间宿舍，

​ //按宿舍号升序挂在已用宿舍链表中。

​ return 宿舍号 //学生退宿舍，删除已用宿舍链表中对应结点，

​ //挂在可用宿舍链表尾部。

​ display_free //输出可用宿舍链表信息。

​ display_used //输出已用宿舍链表信息。

输出

​ display_free依次输出当前可用宿舍链表中的宿舍号，具体格式见样例。

​ display_used依次输出当前已用宿舍链表中的宿舍号，具体格式见样例。

IO模式

本题IO模式为标准输入/输出(Standard IO)，你需要从标准输入流中读入数据，并将答案输出至标准输出流中。

代码：

#include<iostream>
#include<list>
#include<vector>
using namespace std;

class DouNode {
public:
	string name;
	int no;

	DouNode(string na,int n):name(na),no(n) {}
	DouNode(int n) :no(n) {}
	DouNode() {}
	friend bool operator < (DouNode& n1, DouNode& n2);
};

bool operator < (DouNode& n1, DouNode& n2)
{
	if (n1.no < n2.no)
		return 1;
	return 0;
}

int main()
{
	list<DouNode> used_room;
	list<DouNode> free_room;
	list<DouNode>::iterator it;

	int n;
	cin >> n;

	string name;
	int no;
	vector<int> room;
	for (int i = 0; i < n; i++)
	{
		cin >> name >> no;
		room.push_back(no);
		DouNode p(name,no);
		used_room.push_back(p);
	}
	
	int i, j;
	for (i = 101; i < 121; i++)
	{
		for (j = 0; j < room.size(); j++)
		{
			if (room[j] == i)
				break;
		}
		if (j == room.size())
		{
			DouNode p(name, i);
			free_room.push_back(p);
		}
	}

	cin >> n;
	while (n--)
	{
		string op;
		cin >> op;
		if (op == "assign")
		{
			string name;
			cin >> name;
			DouNode t  = free_room.front();
			free_room.pop_front();
			DouNode p(name,t.no);
			used_room.push_back(p);
			used_room.sort();
		}
		else if (op == "return")
		{
			cin >> no;
			for (auto it = used_room.begin(); it != used_room.end(); it++)
			{
				if (it->no == no)
				{
					used_room.erase(it);
					DouNode p(no);
					free_room.push_back(p);
					break;
				}
			}
		}
		else if (op == "display_used")
		{
			
			for (auto it = used_room.begin(); it != used_room.end(); it++)
			{
				if (it != used_room.begin())
					cout << "-";
				cout << it->name << "(" << it->no << ")";
			}
			cout << endl;
		}
		else if (op == "display_free")
		{
			for (auto it = free_room.begin(); it != free_room.end(); it++)
			{
				if (it != free_room.begin())
					cout << "-";
				cout << it->no;
			}
			cout << endl;
		}
	}
	return 0;

}
