import cv2
import numpy as np
import os
#import matplotlib.pyplot as plt

def seg_video(input_path):
    if '_video.' not in input_path:
        print('输入文件错误!')
        return False
    # 打开视频文件
    video = cv2.VideoCapture(input_path)
    # 读取第一帧获取视频的宽度和高度
    video_width = int(video.get(3)) # 编号3表示宽度
    video_height = int(video.get(4)) # 编号4表示高度
    FPS = video.get(cv2.CAP_PROP_FPS)
    ret, frame = video.read()   # ret是bool型，表示是否读取到视频帧，如果未读取到返回false，反之返回true
                                # frame表示每一帧的图像，是个三维矩阵
    if not ret:
        print("无法读取视频帧")
        return False    
    frame = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
    # cv2.cvtColor 是 OpenCV 中的颜色空间转换函数，用于在不同颜色空间之间转换
    # cv2.COLOR_BGR2GRAY 指定将图像从 BGR 颜色空间转换为灰度（GRAYSCALE）颜色空间
    frame = frame[100:video_height, :]  # 裁掉上方区域
    # 截取图像像素行100到底部的图像，将第100行像素行上方的区域截掉，宽度保持不变

    # 使用Sobel算子进行边缘检测
    pframe = cv2.blur(frame, (3, 3))
    # 对灰度图frame进行模糊处理
    # cv2.blur 函数应用均值滤波，将每个像素的值替换为其周围 3x3 区域内像素的平均值，从而平滑图像
    sobelx = cv2.Sobel(pframe, cv2.CV_64F, 1, 0, ksize=3)
    # sobelx 是一个 float64 类型的数组
    # 作用：在水平方向上对模糊后的图像 pframe 进行边缘检测。
    # 解释：cv2.Sobel 函数计算图像的梯度，用于检测边缘。
    # 参数：
    # pframe 是模糊后的灰度图像。
    # cv2.CV_64F 指定输出图像的数据类型为 64 位浮点数，以便保留负值。
    # 1, 0 表示仅计算水平方向的梯度（x 方向），而 y 方向的梯度设为 0。
    # ksize=3 指定 Sobel 算子的卷积核大小为 3。
    sobelx[sobelx < 0] = 0
    # 作用：将 sobelx数组 中的负值设为 0。
    # 解释：水平方向上的 Sobel 边缘检测可能产生负梯度值，将这些负值置零，保留正值以突出边缘强度。
    edges_horizontal = sobelx.astype(np.uint8)
    # 作用：将 sobelx 中的数据从float64位转换为无符号 8 位整型。
    # 解释：将 64 位浮点类型转换为 8 位无符号整型，以便后续处理。这样转换会舍弃小数部分并限定范围为 [0, 255]，可以更好用于表示成灰度图
    nums_list = np.sum(edges_horizontal, axis=0)
    # 作用：计算 edges_horizontal 每一列像素值的总和。
    # 解释：对 edges_horizontal 沿垂直方向求和，结果是一个一维数组 nums_list，
    # 表示图像每一列的边缘强度总和。这在后续用于分析边缘变化，用于检测视频内容的水平分割。
    nums_range = np.max(nums_list) - np.min(nums_list)
    # 计算 nums_list 的取值范围
    nums_normalized = (nums_list - np.min(nums_list)) / nums_range
    # 作用：将 nums_list 归一化到 [0, 1] 的范围。
    # 解释：使用公式 (nums_list - np.min(nums_list)) / nums_range 对数据进行归一化，
    # 将最小值映射为 0，最大值映射为 1。这有助于后续的峰值检测，使算法不受原始值范围的影响。
    peaks, _ = find_peaks(nums_normalized)
    # peaks：是个数组，接受的是 nums_normalized 数组中的局部峰值，表示边缘强度的显著变化位置
    # _：接受的是 peaks 的索引值，即每个峰值在 nums_normalized 中的具体位置。
    block_size = video_width // (len(peaks) + 1)  # 子块大小
    rows = video_height // block_size  # 行数，表示可以分成多少行小块
    columns = video_width // block_size  # 列数，表示可以分成多少列小块
    print(f"行数:{rows},列数:{columns}")

    filename = input_path.split("\\")[-1] # 获得文件的名字
    suffix = "JPG"
    filename = filename.split("."+suffix)[0]

    frame_list = []
    while True:
        ret, frame = video.read()
        if not ret:
            break        
        frame = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
        frame_list.append(frame)
    # 获取视频每一帧的图像

    video.release() # 关闭视频
    data_all = []
    for i in range(rows*columns):   # 对每一个子块进行遍历
        data = []
        x = (i % columns) * block_size  # x表示水平位置
        y = 100 + (i // columns) * block_size   # y表示竖直位置
        # 确定要切割的区域
        start_x = min(max(x, 0), video_width)
        start_y = min(max(y, 0), video_height)
        end_x = min(start_x + block_size, video_width)
        end_y = min(start_y + block_size, video_height)
        # print(f'({start_x},{start_y})({end_x},{end_y})')

        if 'D' in filename:  # date 普通格式
            data.append(filename.split("D")[0])  # Name
            data.append(filename.split("D")[-1].split("_S")[0])  # Date
            if "_S" in filename:
                data.append(filename.split("S")[-1].split("_I")[0])  # S
            if "_I" in filename:
                data.append(filename.split("I")[-1].split("_P_wells")[0])  # I
        else:  # 只有名字
            data.append(filename.split("_")[0])  # Name
            data.append("-")
            data.append("-")
            data.append("-")
        data.append(int(i)+1)  # P_well
        data.append("待检测")  # PB2
        tag = 0

        save_path = r"E:\Embryo_project\img"
        if not os.path.exists(save_path):
            os.makedirs(save_path)
        # 逐帧读取视频并写入输出视频
        for frame in frame_list:
            # 切割指定区域的视频帧
            cropped_frame = frame[start_y:end_y, start_x:end_x]
            tag += 1
            save_path = os.path.join(r"E:\Embryo_project\img", data[1] + '_P_' + str(i + 1) + "_frame_" + str(tag) + '.' + suffix)

            # 如果切割后的帧大小不为 0，则将帧写入输出视频
            if cropped_frame.shape[0] != 0 and cropped_frame.shape[1] != 0: # cropped_frame.shape[0] 表示高度（行数），
                                                                            # cropped_frame.shape[1] 表示宽度（列数）
                cropped_frame = cv2.cvtColor(cropped_frame, cv2.COLOR_GRAY2BGR)

                cv2.imwrite(save_path,cropped_frame)

        data_all.append(data)
        # 释放资源
    return data_all

def find_peaks(vec):
    win_size = len(vec) // 20
    min_peaks_t = 0.8  # 最小峰值阈值
    local_maxima = []
    index = []
    for i in range(win_size, len(vec)-win_size):  # 不考虑边界
        if vec[i] > min_peaks_t:
            win_nums = vec[i - win_size // 2:i + win_size // 2 + 1]
            flag = np.where(win_nums <= vec[i], 0, 1)
            if not np.sum(flag):
                local_maxima.append(vec[i])
                index.append(i)
    return local_maxima, index


def main():
    video_path = input()
    data_all = []
    data_all = seg_video(video_path)
    print(data_all)

if __name__ == "__main__":
    main()
