题目描述

给出一个N*N的迷宫矩阵示意图，从起点[0,0]出发，寻找路径到达终点[N-1, N-1]

要求使用堆栈对象来实现

输入

第一行输入t，表示有t个迷宫

第二行输入n，表示第一个迷宫有n行n列

第三行起，输入迷宫每一行的每个方格的状态，0表示可通过，1表示不可通过

输入n行

以此类推输入下一个迷宫

输出

逐个输出迷宫的路径

如果迷宫不存在路径，则输出no path并回车

如果迷宫存在路径，将路径中每个方格的x和y坐标输出，从起点到终点，每输出四个方格就换行，最终以单词END结尾，具体格式参考示范数据

解题思路

从[0,0]出发，到右下角[n-1,n-1],所以主要步骤是向右和向下，如果两种方式都走不通了再向左，最后向上；

为了防止走回原来走过的路，所以走过的点也将其数值置为1；

若无路可走则退回栈中存储的点，重新规划路线

具体代码

#include<iostream>
#include<stack>
using namespace std;

class Point {
public:
	int x, y;
	Point() {}
	Point(int a, int b) :x(a), y(b) {}
	friend ostream& operator <<(ostream& output, Point& p);
};

ostream& operator <<(ostream& output, Point& p)
{
	output << '[' << p.x << ',' << p.y << ']' << "--";
	return output;
}

int Find_path(stack<Point>& path, int** map,int n,int tag)
{
	map[0][0] = 1;
	int i = 0, j = 0;
	while (!path.empty())
	{
		if (path.top().x == n - 1 && path.top().y == n - 1)
			return 1;
		Point temp = path.top();
		if (temp.y < n - 1  && map[i][j+1] == 0)
		{
			path.push(Point(i, j + 1));
			map[i][j + 1] = 1;
			j++;
		}
		//向下
		else if (temp.x < n - 1  && map[i + 1][j] == 0)
		{
			path.push(Point(i + 1, j));
			map[i + 1][j] = 1;
			i++;
		}
		else if (temp.y > 0  && map[i][j - 1] == 0)
		{
			map[i][j - 1] = 1;
			path.push(Point(i, j-1));
			j--;
		}
		else if (temp.x > 0  && map[i - 1][j] == 0)
		{
			path.push(Point(i - 1, j));
			map[i - 1][j] = 1;
			i--;
		}
		else
		{
			path.pop();
			if (!path.empty())
			{
				i = path.top().x;
				j = path.top().y;
			}
			
		}
	}
	return 0;
}

int main()
{
	int t;
	cin >> t;
	while (t--)
	{
		int n;
		cin >> n;
		int** map = new int* [n];
		for (int i = 0;i < n;i++)
			map[i] = new int[n];

		for (int i = 0;i < n;i++)
		{
			for (int j = 0;j < n;j++)
				cin >> map[i][j];
		}

		stack<Point> path,path1;
		path.push(Point(0, 0));
		if (!Find_path(path, map, n,1))
		{
			cout << "no path" << endl;
			continue;
		}

		while (!path.empty())
		{
			path1.push(path.top());
			path.pop();
		}

		int num = 0;
		while (!path1.empty())
		{
			cout << path1.top();
			num++;
			if (num == 4)
			{
				cout << endl;
				num = 0;
			}
			path1.pop();
		}
		cout << "END" << endl;
	}
	return 0;
}
