【id250】B. 村村通工程（Prim算法）

Nov 20, 2024
2 minute read

题目描述

“村村通"是国家一个系统工程，其包涵有：公路、电力、生活和饮用水、电话网、有线电视网、互联网等等。

村村通公路工程，是国家为构建和谐社会，支持新农村建设的一项重大举措，是一项民心工程。又称“五年千亿元”工程

该工程是指中国力争在5年时间实现所有村庄通沥青路或水泥路，以打破农村经济发展的交通瓶颈，解决9亿农民的出行难题。

现有村落间道路的统计数据表中，列出了有可能建设成标准公路的若干条道路的成本，求使每个村落都有公路连通所需要的最低成本。

要求用Prim算法求解

输入

第1行：顶点数n

第2行：n个顶点编号

第3行：边数m

接着m行：m条边信息，格式为：顶点1 顶点2 权值

最后一行：Prim算法的起点v

输出

第1行：输出最小生成树的权值之和

接着n-1行对应n-1条边信息

按树的生长顺序输出

IO模式

本题IO模式为标准输入/输出(Standard IO)，你需要从标准输入流中读入数据，并将答案输出至标准输出流中。

解题思路

Prime算法，即定义两个数组，一个数组存储挑选出来的节点，然后每一次，把挑选出来的节点与未挑选出来的节点之间的最小边的节点挑选出来，最后生成最小生成树

代码实现

定义两个数组，一个是弧头一个是弧尾，分别存储起点和终点

具体代码

#include<bits/stdc++.h>
#include<unordered_map>
using namespace std;

unordered_map<string, int> h;

class Map
{
private:
	vector<string> point;
	vector<vector<int>> arr;
	vector<int> cost;
	vector<string> prepoint;
	vector<string> head;
	vector<string> tail;
	int start;
	int sum;
	int n;
public:
	Map(int n,vector<string> point, vector<vector<int>> arr,int start)
	{
		this->point = point;
		this->arr = arr;
		this->start = start;
		this->n = n;
		this->sum = 0;

		this->cost.assign(arr[start].begin(), arr[start].end());
		cost[start] = 0;
		vector<string> temp(n,point[start]);
		this->prepoint = temp;
	}

	void prim()
	{
		while (head.size() < n-1)
		{
			int min = INT_MAX;
			int index = 0;
			for (int i = 0; i < cost.size(); i++)
			{
				if (cost[i] < min && cost[i] != 0)
				{
					index = i;
					min = cost[i];
				}
			}

			sum += cost[index];
			head.push_back(point[index]);
			tail.push_back(prepoint[index]);
			cost[index] = 0;

			for (int i = 0; i < cost.size(); i++)
			{
				if (arr[index][i] < cost[i])
				{
					cost[i] = arr[index][i];
					prepoint[i] = point[index];
				}
			}
		}

		out();
	}

	void out()
	{
		cout << sum << endl;
		for (int i = 0; i < n-1; i++)
		{
			cout << tail[i] << " " << head[i] << " " << arr[h[tail[i]]][h[head[i]]] << endl;
		}
	}
};

int main()
{
	int n;
	cin >> n;
	vector<string> point(n);
	for (int i = 0; i < n; i++)
	{
		cin >> point[i];
		h[point[i]] = i;
	}

	vector<vector<int>> arr(n,vector<int>(n,INT_MAX));

	int k;
	cin >> k;
	while (k--)
	{
		string p1, p2;
		int cost;
		cin >> p1 >> p2 >> cost;
		arr[h[p1]][h[p2]] = cost;
		arr[h[p2]][h[p1]] = cost;
	}

	string start;
	cin >> start;
	Map m(n,point, arr, h[start]);

	m.prim();
	return 0;
}
