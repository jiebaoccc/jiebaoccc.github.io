【id305】E. 关于堆的判断

Dec 19, 2024
3 minute read

题目描述

将一系列给定数字顺序插入一个初始为空的小顶堆H[]。随后判断一系列相关命题是否为真。命题分下列几种：

  • x is the root：x是根结点；
  • x and y are siblings：x和y是兄弟结点；
  • x is the parent of y：x是y的父结点；
  • x is a child of y：x是y的一个子结点。

输入

每组测试第1行包含2个正整数N（≤ 1000）和M（≤ 20），分别是插入元素的个数、以及需要判断的命题数。下一行给出区间[−10000,10000]内的N个要被插入一个初始为空的小顶堆的整数。之后M行，每行给出一个命题。题目保证命题中的结点键值都是存在的。

输出

对输入的每个命题，如果其为真，则在一行中输出T，否则输出F。

IO模式

本题IO模式为标准输入/输出(Standard IO)，你需要从标准输入流中读入数据，并将答案输出至标准输出流中。

输入样例

5 4
46 23 26 24 10
24 is the root
26 and 23 are siblings
46 is the parent of 23
23 is a child of 10

输出样例

F
T
F
T

解题思路

首先创建最小堆

x is the root：x是根结点：看是否头结点与x相等

x and y are siblings：x和y是兄弟结点；看两者的索引/2是否相同

x is the parent of y：x是y的父结点； 看(y + 1) / 2 -1 == x

x is a child of y：x是y的一个子结点 看(x + 1) / 2 -1 == y

具体代码

#include<bits/stdc++.h>
using namespace std;

void adjust_deap(vector<int>& num) {
	int len = num.size();

	while (1) {
		int tag = 1;

		for (int i = len / 2 - 1; i >= 0; i--) {
			int a = INT_MAX, b = INT_MAX;

			if ((i + 1) * 2 - 1 < len) {
				a = num[(i + 1) * 2 - 1];
			}
			if ((i + 1) * 2 < len) {
				b = num[(i + 1) * 2];
			}

			int min_index = a < b ? (i + 1) * 2 - 1 : (i + 1) * 2;
			if (num[i] > num[min_index]) {
				swap(num[i], num[min_index]);
				tag = 0;
			}
		}

		if (tag)break;
	}
	return;

}

int main() {
	int N, M; cin >> N >> M;
	vector<int> num(N);
	for (int i = 0; i < N; i++)cin >> num[i];
	adjust_deap(num);

	for (int j = 0; j < M; j++) {
		int n; cin >> n;
		string str; 
		getline(cin,str);
		
		if (str.find("root") != -1) {
			if (n == num[0]) cout << "T" << endl;
			else cout << "F" << endl;
		}

		else if (str.find("and") != -1) {
			int m = (str[5] - '0') * 10 + (str[6] - '0');

			int x = -1, y = -1;
			for (int i = 0; i < N; i++) {
				if (num[i] == n) x = i;
				if (num[i] == m) y = i;
			}

			if ((x + 1) / 2 == (y + 1) / 2) cout << "T" << endl;
			else cout << "F" << endl;
		}

		else if (str.find("parent") != -1) {
			int len = str.size();
			int m = (str[len - 2] - '0') * 10 + (str[len - 1] - '0');

			int x = -1, y = -1;
			for (int i = 0; i < N; i++) {
				if (num[i] == n) x = i;
				if (num[i] == m) y = i;
			}

			if ((y + 1) / 2 - 1 == x) cout << "T" << endl;
			else cout << "F" << endl;
		}

		else {
			int len = str.size();
			int m = (str[len - 2] - '0') * 10 + (str[len - 1] - '0');

			int x=-1, y=-1;
			for (int i = 0; i < N; i++) {
				if (num[i] == n) x = i;
				if (num[i] == m) y = i;
			}

			if ((x + 1) / 2 -1 == y) cout << "T" << endl;
			else cout << "F" << endl;
		}
	}

	return 0;
}
