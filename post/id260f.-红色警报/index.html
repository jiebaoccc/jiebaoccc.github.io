【id260】F. 红色警报

Nov 20, 2024
2 minute read

题目描述

战争中保持各个城市间的连通性非常重要。本题要求你编写一个报警程序，当失去一个城市导致国家被分裂为多个无法连通的区域时，就发出红色警报。注意：若该国本来就不完全连通，是分裂的k个区域，而失去一个城市并不改变其他城市之间的连通性，则不要发出警报。

输入

输入在第一行给出两个整数N（0 < N ≤ 500）和M（≤ 5000），分别为城市个数（于是默认城市从0到N-1编号）和连接两城市的通路条数。随后M行，每行给出一条通路所连接的两个城市的编号，其间以1个空格分隔。在城市信息之后给出被攻占的信息，即一个正整数K和随后的K个被攻占的城市的编号。

注意：输入保证给出的被攻占的城市编号都是合法的且无重复，但并不保证给出的通路没有重复。

输出

对每个被攻占的城市，如果它会改变整个国家的连通性，则输出Red Alert: City k is lost!，其中k是该城市的编号；否则只输出City k is lost.即可。如果该国失去了最后一个城市，则增加一行输出Game Over.。

IO模式

本题IO模式为标准输入/输出(Standard IO)，你需要从标准输入流中读入数据，并将答案输出至标准输出流中

输入样例1

5 4
0 1
1 3
3 0
0 4
5
1 2 0 4 3

输出样例1

City 1 is lost.
City 2 is lost.
Red Alert: City 0 is lost!
City 4 is lost.
City 3 is lost.
Game Over.

解题思路

通过DFS遍历图，找出有几个连通分量，如果连通分量比之前的多，则发出红色警报

具体代码

#include<bits/stdc++.h>
using namespace std;

void DFS(vector<vector<int>> arr, vector<bool>& Bool, int index)
{
	for (int i = 0; i < arr.size(); i++)
	{
		if (arr[index][i] == 1 && Bool[i] == false)
		{
			Bool[i] = true;
			DFS(arr,Bool,i);
		}
	}
}

int Judge(vector<vector<int>> arr, int N,vector<bool> Bool)
{
	int ans = 0;
	for (int i = 0; i < N; i++)
	{
		if (Bool[i] == false)
		{
			Bool[i] = true;
			DFS(arr, Bool, i);
			ans++;
		}
	}
	return ans;
}

int main()
{
	int N, M;
	cin >> N >> M;

	vector<vector<int>> arr(N,vector<int>(N,0));

	while (M--)
	{
		int i, j;
		cin >> i >> j;
		arr[i][j] = 1;
		arr[j][i] = 1;
	}
	vector<bool> Bool(N,false);
	int num = Judge(arr, N,Bool);

	int k;
	cin >> k;
	while (k--)
	{
		int n;
		cin >> n;

		Bool[n] = true;

		for (int i = 0; i < N; i++)
		{
			arr[n][i] = 0;
			arr[i][n] = 0;
		}

		int new_num = Judge(arr,N,Bool);
		if (new_num > num)
		{
			num = new_num;
			cout << "Red Alert: City " << n << " is lost!" << endl;
		}
		else
		{
			num = new_num;
			cout << "City " << n << " is lost." << endl;
		}

		int tag = 0;
		for (int i = 0; i < N; i++)
		{
			if (Bool[i] == false)
			{
				tag = 1;
				break;
			}
		}
		if (tag)
			continue;
		cout << "Game Over." << endl;
	}

	return 0;
}
