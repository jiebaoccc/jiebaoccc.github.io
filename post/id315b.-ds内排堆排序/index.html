【id315】B. DS内排—堆排序

Dec 19, 2024
2 minute read

题目描述

给定一组数据，使用堆排序完成数据的降序排序。（建小顶堆)。

输入

数据个数n，n个整数数据

输出

初始创建的小顶堆序列

每趟交换、筛选后的数据序列，输出格式见样例

IO模式

本题IO模式为标准输入/输出(Standard IO)，你需要从标准输入流中读入数据，并将答案输出至标准输出流中。

输入样例

8 34 23 677 2 1 453 3 7

输出样例

8 1 2 3 7 23 453 677 34
8 2 7 3 34 23 453 677 1
8 3 7 453 34 23 677 2 1
8 7 23 453 34 677 3 2 1
8 23 34 453 677 7 3 2 1
8 34 677 453 23 7 3 2 1
8 453 677 34 23 7 3 2 1
8 677 453 34 23 7 3 2 1

解题思路

从最后一个有孩子的节点开始，依次向前遍历，检验是否符合最小堆的要求，循环遍历，直到不需要发生交换

最后一个有孩子的节点的索引为 len/2-1 , 其孩子节点为 (i+1)* 2-1，(i+1)* 2

具体代码

#include<bits/stdc++.h>
using namespace std;

void display(vector<int> num) {
	cout << num.size() << " ";
	for (int i = 0; i < num.size(); i++) {
		cout << num[i];
		if (i < num.size() - 1)cout << " ";
		else cout << endl;
	}
}

void adjust_heap(vector<int>& num,int n) {
	while (1) {
		int tag = 1;
		for (int i = n / 2 - 1; i >= 0; i--) {
			int a = INT_MAX, b = INT_MAX;
			int min_index = 0;
			if ((i + 1) * 2 - 1 < n) {
				a = num[(i + 1) * 2 - 1];
			}
			if ((i + 1) * 2 < n) {
				b = num[(i + 1) * 2];
			}

			min_index = a < b ? (i + 1) * 2 - 1 : (i + 1) * 2;
			if (num[i] > num[min_index]) {
				swap(num[i], num[min_index]);
				tag = 0;
			}
		}
		if (tag) break;
	}
}


void sort_heap(vector<int>& num,int n) {
	for (int i = n - 1; i > 0; i--) {
		swap(num[0],num[i]);
		adjust_heap(num,i);
		display(num);
	}
}

int main() {
	int n; cin >> n;
	vector<int> num(n);
	int min_ = INT_MAX;
	for (int i = 0; i < n; i++) {
		cin >> num[i];
		min_ = min(min_, num[i]);
	}

	adjust_heap(num, n);
	display(num);
	sort_heap(num, n);
}
