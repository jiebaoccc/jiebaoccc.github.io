【id:249】C. 村村通工程（Kruskal算法）

Nov 20, 2024
2 minute read

题目描述

“村村通"是国家一个系统工程，其包涵有：公路、电力、生活和饮用水、电话网、有线电视网、互联网等等。

村村通公路工程，是国家为构建和谐社会，支持新农村建设的一项重大举措，是一项民心工程。又称“五年千亿元”工程

该工程是指中国力争在5年时间实现所有村庄通沥青路或水泥路，以打破农村经济发展的交通瓶颈，解决9亿农民的出行难题。

现有村落间道路的统计数据表中，列出了有可能建设成标准公路的若干条道路的成本，求使每个村落都有公路连通所需要的最低成本。

要求用Kruskal算法求解

输入

第1行：顶点数n

第2行：n个顶点编号

第3行：边数m

接着m行：m条边信息，格式为：顶点1 顶点2 权值

输出

第1行：输出最小生成树的权值之和

接着n-1行对应n-1条边信息

如果能找到最小生成树，按树的生长顺序输出, 边顶点按数组序号升序输出

如果输入数据不足以保证畅通，则直接输出−1，无需输出任何边信息

IO模式

本题IO模式为标准输入/输出(Standard IO)，你需要从标准输入流中读入数据，并将答案输出至标准输出流中。

解题思路

并查集，首先根据权重进行排序，然后每一次插入时通过并查集判断是否存在回环，如果存在则跳过这条边，如果不存在则插入

具体代码

#include<bits/stdc++.h>
#include<unordered_map>
using namespace std;

unordered_map<string, int> h;

class Union_find
{
private:
	vector<int> parent;
	vector<int> rank;
	int len;
public:
	Union_find() {}
	Union_find(vector<vector<int>>& edges,int n)
	{
		len = n;
		parent.assign(len+1,-1);
		rank.assign(len+1,0);
	}
	int Find(int x)
	{
		if (parent[x] == -1)
			parent[x] = x;
		
		if (parent[x] != x)
		{
			parent[x] = Find(parent[x]);
		}
		return parent[x];
	}
	void Union(int x, int y)
	{
		int p1 = Find(x);
		int p2 = Find(y);

		if (rank[p1] < rank[p2])
		{
			parent[p1] = p2;
		}
		else if (rank[p2] < rank[p1])
			parent[p2] = p1;
		else
		{
			parent[p2] = p1;
			rank[p1]++;
		}
	}
};

class Map
{
private:
	vector<vector<int>> edges;
	vector<string> point;
	vector<vector<int>> ans;
	int sum;
	int n;
public:
	Map(int n,vector<vector<int>> edges, vector<string> point)
	{
		this->edges = edges;
		this->point = point;
		this->sum = 0;
		this->n = n;
	}

	void Kruskal()
	{
		int i = 0;
		sum += edges[i][2];
		ans.push_back(edges[i++]);


		while (ans.size() < n - 1)
		{
			if (i == edges.size())
			{
				cout << -1 << endl;
				return;
			}
			vector<vector<int>> tem = ans;
			tem.push_back(edges[i]);

			int tag = 1;
			Union_find union_find(tem,n);
			for (auto& edge : tem)
			{
				int p1 = edge[0], p2 = edge[1];
				if (union_find.Find(p1) != union_find.Find(p2))
					union_find.Union(p1, p2);
				else
				{
					tag = 0;
					break;
				}
			}

			if (tag)
			{
				ans = tem;
				sum += edges[i++][2];
			}
			else
				i++;
		}
		out();
	}

	void out()
	{
		cout << sum << endl;
		if (ans.empty())
		{
			cout << -1<<endl;
			return;
		}
		for (auto a : ans)
		{
			if (a[0] > a[1])
			{
				swap(a[0], a[1]);
			}
			cout << point[a[0]] << " " << point[a[1]] << " " << a[2] << endl;
		}
	}

};

int cmp(vector<int> a, vector<int> b)
{
	return a[2] < b[2];
}

int main()
{
	int n;
	cin >> n;
	vector<string> point(n);

	for (int i = 0; i < n; i++)
	{
		cin >> point[i];
		h[point[i]] = i;
	}

	vector<vector<int>> edges;

	int k;
	cin >> k;
	while (k--)
	{
		vector<int> edge;
		string p1, p2;
		int cost;

		cin >> p1 >> p2 >> cost;
		edge.push_back(h[p1]);
		edge.push_back(h[p2]);
		edge.push_back(cost);

		edges.push_back(edge);
	}

	sort(edges.begin(),edges.end(), cmp);

	Map m (n,edges,point);
	m.Kruskal();

	return 0;
}
